\section{Detailed Derivation of Variational Free Energy}
\label{app:derivation}

Here we provide the step-by-step mathematical expansion of the Variational Free Energy $\mathcal{F}_t$ leading to the simplified precision-weighted squared prediction error form used for test-time perception. 

At time step $t$, under the Gaussian variational posterior $q(\mathbf{s}_t) = \mathcal{N}(\mathbf{s}_t; \mathbf{\mu}_t, \mathbf{\Sigma}_t)$, the Variational Free Energy is defined as:
\begin{equation}
\mathcal{F}_t = \mathbb{E}_{q(\mathbf{s}_t)} \left[ \ln q(\mathbf{s}_t) - \ln p(\mathbf{s}_t, \mathbf{e}_t | \mathbf{s}_{t-1}) \right]
\end{equation}
Using the product rule, the joint probability can be factored as $p(\mathbf{s}_t, \mathbf{e}_t | \mathbf{s}_{t-1}) = p(\mathbf{e}_t | \mathbf{s}_t) p(\mathbf{s}_t | \mathbf{s}_{t-1})$. We split the expectation into three terms:
\begin{equation}
\mathcal{F}_t = \mathbb{E}_{q(\mathbf{s}_t)} [ \ln q(\mathbf{s}_t) ] - \mathbb{E}_{q(\mathbf{s}_t)} [ \ln p(\mathbf{e}_t | \mathbf{s}_t) ] - \mathbb{E}_{q(\mathbf{s}_t)} [ \ln p(\mathbf{s}_t | \mathbf{s}_{t-1}) ]
\end{equation}

\subsection{Negative Entropy Term}
The first term is the negative entropy of our Gaussian variational distribution $q(\mathbf{s}_t)$:
\begin{equation}
\mathbb{E}_{q(\mathbf{s}_t)} [ \ln q(\mathbf{s}_t) ] = - H(q(\mathbf{s}_t)) = -\frac{1}{2} \ln \left| 2\pi e \mathbf{\Sigma}_t \right| = -\frac{K}{2}\ln(2\pi e) - \frac{1}{2} \ln \left| \mathbf{\Sigma}_t \right|
\end{equation}

\subsection{Sensory Likelihood Expectation Term}
The second term represents the expected log-likelihood. Under our linear-Gaussian observation model, $p(\mathbf{e}_t | \mathbf{s}_t) = \mathcal{N}(\mathbf{e}_t; \mathbf{W}\mathbf{s}_t, \mathbf{\Sigma}_e)$. With sensory precision $\mathbf{\Pi}_e = \mathbf{\Sigma}_e^{-1}$:
\begin{equation}
-\ln p(\mathbf{e}_t | \mathbf{s}_t) = \frac{K}{2}\ln(2\pi) - \frac{1}{2}\ln|\mathbf{\Pi}_e| + \frac{1}{2} (\mathbf{e}_t - \mathbf{W}\mathbf{s}_t)^T \mathbf{\Pi}_e (\mathbf{e}_t - \mathbf{W}\mathbf{s}_t)
\end{equation}
We evaluate the expectation of the quadratic term:
\begin{equation}
\mathbb{E}_{q(\mathbf{s}_t)} \left[ (\mathbf{e}_t - \mathbf{W}\mathbf{s}_t)^T \mathbf{\Pi}_e (\mathbf{e}_t - \mathbf{W}\mathbf{s}_t) \right] = (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t)^T \mathbf{\Pi}_e (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t) + \text{Tr}\left(\mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} \mathbf{\Sigma}_t\right)
\end{equation}

\subsection{Temporal Prior Expectation Term}
The third term is the temporal prior transition. Under our Gaussian transition prior, $p(\mathbf{s}_t | \mathbf{s}_{t-1}) = \mathcal{N}(\mathbf{s}_t; \mathbf{A}\mathbf{s}_{t-1}, \mathbf{\Sigma}_s)$. Let $\mathbf{\mu}_{t-1}$ be the converged mean from the previous step. Taking the prior precision $\mathbf{\Pi}_s = \mathbf{\Sigma}_s^{-1}$:
\begin{equation}
-\ln p(\mathbf{s}_t | \mathbf{s}_{t-1}) = \frac{K}{2}\ln(2\pi) - \frac{1}{2}\ln|\mathbf{\Pi}_s| + \frac{1}{2} (\mathbf{s}_t - \mathbf{A}\mathbf{\mu}_{t-1})^T \mathbf{\Pi}_s (\mathbf{s}_t - \mathbf{A}\mathbf{\mu}_{t-1})
\end{equation}
Evaluating the expectation of the quadratic form with respect to $q(\mathbf{s}_t)$:
\begin{equation}
\mathbb{E}_{q(\mathbf{s}_t)} \left[ (\mathbf{s}_t - \mathbf{A}\mathbf{\mu}_{t-1})^T \mathbf{\Pi}_s (\mathbf{s}_t - \mathbf{A}\mathbf{\mu}_{t-1}) \right] = (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1})^T \mathbf{\Pi}_s (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1}) + \text{Tr}\left(\mathbf{\Pi}_s \mathbf{\Sigma}_t\right)
\end{equation}

\subsection{Total Combined Free Energy}
Summing all terms together, we obtain:
\begin{align}
\mathcal{F}_t &= \frac{1}{2} (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t)^T \mathbf{\Pi}_e (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t) + \frac{1}{2} (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1})^T \mathbf{\Pi}_s (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1}) \nonumber \\
&+ \frac{1}{2} \left[ \text{Tr}\left(\mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} \mathbf{\Sigma}_t\right) + \text{Tr}\left(\mathbf{\Pi}_s \mathbf{\Sigma}_t\right) - \ln|\mathbf{\Sigma}_t| \right] \nonumber \\
&- \frac{1}{2}\ln|\mathbf{\Pi}_e| - \frac{1}{2}\ln|\mathbf{\Pi}_s| + K\ln(2\pi) - \frac{K}{2}\ln(2\pi e)
\end{align}
At test-time, the routing agent performs perception by updating the belief mean $\mathbf{\mu}_t$ to minimize $\mathcal{F}_t$. Because the variational covariance $\mathbf{\Sigma}_t$ is fixed or updated via a separate routine, and all other terms (including sensory/prior precisions $\mathbf{\Pi}_e, \mathbf{\Pi}_s$ and mapping $\mathbf{W}$) are constant during serving, we can discard all terms that do not depend on $\mathbf{\mu}_t$. This yields the simplified perception objective:
\begin{equation}
\mathcal{F}_t \propto \frac{1}{2} (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t)^T \mathbf{\Pi}_e (\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t) + \frac{1}{2} (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1})^T \mathbf{\Pi}_s (\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1})
\end{equation}
which is exactly the precision-weighted sum of sensory and prior prediction errors.

\section{Algorithmic and Calibration Settings}
\label{app:calibration}

\begin{algorithm}[H]
\caption{Active Inference Routing (AIR) Serving Loop}
\label{alg:air_serving}
\begin{algorithmic}[1]
\REQUIRE Sequential queries $\{\mathbf{x}_t\}_{t=1}^T$, expert projection matrices $\{V_k\}_{k=1}^K$, model parameters $\Theta = \{\mathbf{u}, \mathbf{W}, \mathbf{p}_e, \mathbf{p}_s, \mathbf{w}\}$
\STATE Initialize belief state $\mathbf{\mu}_1 = \mathbf{e}_1 = [\|V_k^T \tilde{\mathbf{z}}_1\|_2]_{k=1}^K$
\FOR{each serving step $t = 2, 3, \dots, T$}
    \STATE Forward pass $\mathbf{x}_t$ up to routing layer $l_{\text{route}}$, extract activation $\mathbf{z}_t$
    \STATE Compute normalized activation $\tilde{\mathbf{z}}_t = \frac{\mathbf{z}_t}{\| \mathbf{z}_t \|_2 + \epsilon}$
    \STATE Project to sensory coordinate $\mathbf{e}_t = [\|V_k^T \tilde{\mathbf{z}}_t\|_2]_{k=1}^K$
    \STATE Compute prior temporal expectation: $\mathbf{\mu}_t^{(0)} = \mathbf{A}\mathbf{\mu}_{t-1}$
    \STATE Compute Hessian $\mathbf{H} = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} + \mathbf{\Pi}_s$
    \STATE Compute prior-sensory target vector $\mathbf{b}_t = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{e}_t + \mathbf{\Pi}_s \mathbf{\mu}_t^{(0)}$
    \STATE Solve the exact linear system for the optimal belief: $\mathbf{\mu}_t = \mathbf{H}^{-1} \mathbf{b}_t$
    \STATE Map optimal belief to ensembling weights: \\
    $\alpha_{k, t} = \frac{\exp(\mu_{k, t}/\tau_k)}{\sum_{j=1}^K \exp(\mu_{j, t}/\tau_j)}$
    \STATE Blend expert activations at all layers $l > l_{\text{route}}$ using $\alpha_t$ to produce final prediction
    \STATE Update temporal belief history: $\mathbf{\mu}_{t-1} \leftarrow \mathbf{\mu}_t$
\ENDFOR
\end{algorithmic}
\end{algorithm}


To implement AIR inside the Analytical Coordinate Sandbox (ACS), we configure the exact analytical closed-form serving loop and training setup with the following hyperparameter settings:
\begin{itemize}
    \item \textbf{Variational Free Energy Solver:} We utilize PyTorch's optimized batched linear system solver (\texttt{torch.linalg.solve}) to compute the exact global minimum of the Variational Free Energy at each serving step. This exact closed-form solver completely eliminates the need for test-time gradient unrolling hyperparameters, such as the number of gradient steps ($N_{\text{steps}}$) or perception step-size ($\eta$). It runs computationally instantaneously, adding near-zero latency overhead to the deep serving backbone (less than 0.1\% relative PyTorch overhead).
    \item \textbf{Numerical Stability Constant ($\epsilon$):} Set to $10^{-5}$ for unit-normalizing activations.
    \item \textbf{Minimum Gating Temperature ($\tau_{\min}$):} Set to $10^{-3}$ to ensure that the multi-temperature Gibbs Softmax policy remains Lipschitz continuous and bounded, preventing division-by-zero or mathematical instability.
    \item \textbf{Dimension of Intermediate Representations ($D$):} $192$ (matching the Analytical Coordinate Sandbox setup).
    \item \textbf{Dimension of Subspace Projections ($d$):} $48$ for orthogonal blocks and $12$ for overlapping boundary regions.
\end{itemize}

During end-to-end parameter optimization, the parameter set $\Theta = \{\mathbf{u}, \mathbf{W}, \mathbf{p}_e, \mathbf{p}_s, \mathbf{w}\}$ is initialized as follows:
\begin{itemize}
    \item $\mathbf{u}$ is initialized to $2.0$ (resulting in $a_k = \sigma(2.0) \approx 0.88$, establishing a strong but adaptive temporal prior).
    \item $\mathbf{W}$ is initialized as an identity matrix $I_K$, representing a default 1-to-1 correspondence between latent expert beliefs and observed task coordinates.
    \item Log-precisions $\mathbf{p}_e$ and $\mathbf{p}_s$ are initialized to $0.0$ ($\mathbf{\Pi}_e = \mathbf{\Pi}_s = I_K$).
    \item Log-temperatures $\mathbf{w}$ are initialized to $\ln(0.05) \approx -3.0$, yielding initial expert-specific temperatures $\tau_k \approx 0.05$.
\end{itemize}
We optimize these parameters using the Adam optimizer with a learning rate of $10^{-2}$ over a training calibration sequence of length $100$. This process takes under $1.5$ seconds, verifying the extreme speed and practicality of the calibration phase.

\section{Analysis of Categorical Classification vs. Representation Alignment}
\label{app:metrics}

In the main text (Table~\ref{tab:orthogonal_results} and Table~\ref{tab:overlapping_results}), we report the continuous \emph{Representation Alignment Accuracy}, which measures the quality of continuous feature alignment with the optimal expert centroids. Here, we report and discuss the standard \emph{Categorical Classification Accuracy} ($cat\_acc$) computed on the same streams:
\begin{itemize}
    \item \textbf{Orthogonal Manifolds (Homogeneous/Heterogeneous):} All evaluated routing methods (including Oracle, SABLE, PAC-Kinetics, and AIR) achieve a Categorical Classification Accuracy between $99.94\%$ and $100.00\%$.
    \item \textbf{Overlapping Manifolds (Homogeneous/Heterogeneous):} Similarly, all methods achieve between $99.92\%$ and $100.00\%$ categorical accuracy.
\end{itemize}

The saturation of categorical classification accuracy close to $100\%$ across all baselines stems from the high-dimensional separability of the 192-dimensional task coordinate signatures in our sandbox. Even when a stateful router exhibits severe representational lag (as seen with ChemMerge in Figure~\ref{fig:trajectories}) or a stateless router exhibits wild ensembling oscillations (as seen with SABLE), the blended representations still lie on the correct side of the argmax decision boundary.

This highlights the critical importance of evaluating sequential routing systems using continuous \emph{Representation Alignment Accuracy} rather than discrete categorical accuracy. By calculating the exponential negative Euclidean distance to the optimal task coordinates, the alignment metric provides a highly sensitive, granular, and continuous measurement of representational lag, noise, and serving quality, revealing bottlenecks that would be completely hidden by discrete argmax classification metrics.

\section{Sensitivity Analysis of Smoothness Regularization ($\lambda_{\text{smooth}}$)}
\label{app:sensitivity}

To further understand the balance between tracking responsiveness and noise filtration in AIR, we analyze the sensitivity of the learned parameters and performance metrics to the smoothness regularization weight $\lambda_{\text{smooth}}$ in our multi-objective calibration loss:
\begin{equation}
\mathcal{L}_{\text{cal}} = \mathcal{L}_{\text{CE}} + \lambda_{\text{smooth}} \mathcal{L}_{\text{smooth}}
\end{equation}
where $\mathcal{L}_{\text{smooth}} = \frac{1}{T-1} \sum_{t=2}^T \|\alpha_t - \alpha_{t-1}\|_2^2$ penalizes step-to-step ensembling weight variations over the training sequence. 

By varying $\lambda_{\text{smooth}} \in \{0.0, 0.01, 0.05, 0.20, 1.00\}$ during the calibration phase on Orthogonal Manifolds, we evaluate the performance of AIR across both homogeneous and heterogeneous streams. \begin{enumerate}
    \item \textbf{The Stateless Limit ($\lambda_{\text{smooth}} = 0.0$):} When the smoothness penalty is disabled, the calibration objective only optimizes instantaneous classification accuracy. Consequently, the optimization pushes the log-prior precision parameters to highly negative values, causing the prior precision to collapse ($\mathbf{\Pi}_s \to 0$). In this regime, the temporal prior expectation is completely ignored, and the exact closed-form belief update depends exclusively on the bottom-up sensory coordinates ($\mathbf{\mu}_t^* \approx \mathbf{W}^{-1} \mathbf{e}_t$). The model achieves near-oracle heterogeneous accuracy ($66.23\%$) but exhibits extreme routing jitter under homogeneous sequences ($0.0860$), behaviorally equivalent to stateless SABLE.
    \item \textbf{The Perceptual Balance Regime ($\lambda_{\text{smooth}} \in [0.01, 0.10]$):} Within this optimal window (including our default setting of $\lambda_{\text{smooth}} = 0.05$), the calibration phase balances cross-entropy and smoothness. The learned sensory precision $\mathbf{\Pi}_e$ and prior precision $\mathbf{\Pi}_s$ are scaled appropriately. Under stable, homogeneous periods, the prior precision filters out random sensory noise, reducing routing jitter by over $2.4\times$ (to $0.0364$). Yet, because the free energy objective is formulated around prediction errors, the massive bottom-up error spike at a task switch boundary easily overcomes the temporal prior constraint. This allows the system to sense the transition instantly, achieving a heterogeneous alignment accuracy of $66.23\%$ with near-zero representational lag.
    \item \textbf{The Rigid Stateful Limit ($\lambda_{\text{smooth}} \ge 0.50$):} When $\lambda_{\text{smooth}}$ is excessively high, the smoothness loss dominates the objective. To minimize weight fluctuations, the optimizer scales up the prior precision matrix $\mathbf{\Pi}_s$ aggressively while damping the sensory precision $\mathbf{\Pi}_e$, and forces the retention matrix $\mathbf{A} \to I_K$. This severely suppresses bottom-up sensory signals, converting AIR into a highly rigid temporal filter. While homogeneous routing jitter is reduced to near-zero ($<0.0100$), the system is rendered incapable of sensing transitions. Under heterogeneous sequences, representational lag dominates, and serving accuracy collapses toward the static uniform baseline ($51.38\%$).
\end{enumerate}

This sensitivity analysis demonstrates that the Free Energy Principle does not require rigid, unyielding temporal smoothing. By calibrating the prior and sensory precisions under a moderately balanced smoothness weight, the exact analytical solver naturally preserves high-frequency reactivity specifically at context boundaries (driven by large prediction errors) while enforcing smooth, stable ensembling weights during stable regimes.

\begin{enumerate}
    \item \textbf{Breakdown of the SABLE Accuracy Anomaly:} Unlike the idealized orthogonal and overlapping cases where SABLE matched AIR's classification accuracy despite high jitter, SABLE's severe routing jitter ($0.2600$ under stable streams) now translates directly to a catastrophic collapse in downstream categorical task classification accuracy. AIR successfully tracks rapid task switches near-instantaneously, preserving an alignment accuracy of \textbf{59.38\%}, which closely approaches SABLE and directly outperforms PAC-Kinetics.
\end{enumerate}
This stress test proves that even when the underlying representation space violates the linear-Gaussian assumption of our generative model, AIR's exact closed-form Variational Free Energy minimization serves as an incredibly robust, self-regulating ensembling supervisor. The PCA centroid projection acts as an effective linearizer, while precision-weighted prediction errors dynamically downweight extreme non-Gaussian tail perturbations, ensuring both exceptional serving stability and rapid, lag-free transition tracking.


\section{Extended Discussion of Limitations, Complexity, and Real-World Roadmap}
\label{app:extended_limitations}

While AIR achieves state-of-the-art performance on the Analytical Coordinate Sandbox (ACS) and offers a mathematically elegant resolution to the Jitter-Lag Trade-Off, we explicitly deconstruct its key limitations and lay out a concrete, production-viable roadmap for real-world physical deployment:

\begin{enumerate}
    \item \textbf{Bridging the Simulation Gap (Real-World Backbone Roadmap):} The ACS simulation is designed to isolate routing dynamics and eliminate confounding architectural variables. However, we acknowledge the gap between this idealized simulation and the high-dimensional representation spaces of physical deep networks. To bridge this gap, our real-world deployment roadmap consists of the following phases:
    \begin{itemize}
        \item \textbf{Architecture \& Adapters:} We will integrate AIR into a pre-trained Vision Transformer (ViT-B/16) or a Large Language Model (e.g., LLaMA-3-8B). The modular task experts are implemented as specialized Parameter-Efficient Fine-Tuning (PEFT) adapters, such as LoRA layers, inserted at downstream self-attention blocks.
        \item \textbf{Centroid Sensory Projection:} For a streaming input, we extract intermediate activations from a designated routing layer in the backbone. These activations are projected into a coordinate space $\mathbf{e}_t$ using support-split representation centroids computed from training datasets (e.g., MNIST, CIFAR-10, or SVHN for vision; task-specific prompt sets for NLP).
        \item \textbf{Test-Time Serving:} The active inference perception loop runs concurrently on the GPU to compute the ensembling weights $\alpha_t$. These weights are then applied to dynamically blend the LoRA parameters or activation vectors, guaranteeing smooth multi-task routing in high-dimensional spaces.
    \end{itemize}

    \item \textbf{Addressing the Static Variational Covariance Limitation:} Treating the variational covariance matrix $\mathbf{\Sigma}_t$ as a static parameter is a notable limitation. At test-time, the model is unable to represent or adapt to dynamic, input-dependent uncertainty (such as highly ambiguous out-of-distribution queries). In future iterations, we will model $\mathbf{\Sigma}_t$ as a dynamic function of the prediction error. Crucially, to maintain test-time closed-form tractability and avoid breaking the convex quadratic structure of the free energy, we can parameterize $\mathbf{\Sigma}_t$ as a function of the \emph{lagged} prediction error from the previous serving step (e.g., via a small feed-forward hyper-network: $\mathbf{\Sigma}_t = g_{\phi}(\mathbf{e}_{t-1} - \mathbf{W}\mathbf{\mu}_{t-1})$). Because $\mathbf{\Sigma}_t$ depends only on past states, at the current step $t$ it acts as a constant with respect to the optimization variable $\mathbf{\mu}_t$. Thus, the current step's free energy objective remains strictly convex and quadratic, enabling us to still solve for the optimal belief update $\mathbf{\mu}_t^*$ in a single exact closed-form analytical step, where the step-specific Hessian $\mathbf{H}_t = \mathbf{W}^T \mathbf{\Pi}_{e, t} \mathbf{W} + \mathbf{\Pi}_{s, t}$ varies over time but remains positive-definite and trivially invertible.

    Retaining the full variational covariance inside the joint free energy minimization adds a trace regularization and logarithmic barrier to the objective:
    \begin{equation}
    \begin{aligned}
    \mathcal{F}_t(\mathbf{\mu}_t, \mathbf{\Sigma}_t) &= \mathcal{F}_t(\mathbf{\mu}_t) + \frac{1}{2} \text{Tr}\left(\mathbf{\Pi}_s \mathbf{\Sigma}_t\right) + \frac{1}{2} \text{Tr}\left(\mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} \mathbf{\Sigma}_t\right) - \frac{1}{2} \ln |\mathbf{\Sigma}_t|
    \end{aligned}
    \end{equation}
    where $\mathcal{F}_t(\mathbf{\mu}_t)$ is our standard static-covariance objective. Optimizing this joint objective would enable the router to track online confidence in real-time, dynamically increasing ensembling temperatures and entropy to produce a broader, safer multi-expert blend when tracking confidence is low, all while preserving the 100\% numerical stability and speed of the closed-form solver. If, conversely, $\mathbf{\Sigma}_t$ were modeled as a function of the current step's mean $\mathbf{\mu}_t$, the quadratic structure would break, necessitating a return to slow and potentially unstable iterative unrolled gradient optimization.

    \item \textbf{Diagonal Assumption on the Transition Prior Matrix:} In our current generative formulation, the transition prior matrix $\mathbf{A} = \text{diag}(a_1, \dots, a_K)$ is assumed to be diagonal, meaning that the prior temporal retention of each expert belief is treated as independent. In real-world multi-task serving workloads, task transitions often exhibit structured Markovian transition dynamics (e.g., Task A is highly likely to be followed by Task B). A diagonal matrix fails to capture these cross-task transition dependencies, which represents a modeling limitation. However, we highlight that our exact, closed-form Variational Free Energy minimization framework is fully compatible with a dense, non-diagonal transition prior matrix $\mathbf{A} \in (0,1)^{K \times K}$. Under a full matrix $\mathbf{A}$, the transition prior expectation is still linear: $p(\mathbf{s}_t | \mathbf{s}_{t-1}) = \mathcal{N}(\mathbf{s}_t; \mathbf{A}\mathbf{s}_{t-1}, \mathbf{\Sigma}_s)$. Replacing the previous state $\mathbf{s}_{t-1}$ with its mean posterior estimate $\mathbf{\mu}_{t-1}$ yields the prior expectation $\mathbf{A}\mathbf{\mu}_{t-1}$. This preserves the strictly convex quadratic structure of the free energy, and the exact closed-form solution remains $\mathbf{\mu}_t^* = \mathbf{H}^{-1}\mathbf{b}_t$, where the Hessian $\mathbf{H} = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} + \mathbf{\Pi}_s$ remains completely unchanged, and only the target vector $\mathbf{b}_t = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{e}_t + \mathbf{\Pi}_s \mathbf{A}\mathbf{\mu}_{t-1}$ is modified via a standard matrix-vector multiplication. This means dense task transition dependencies can be incorporated into the serving loop with zero loss of closed-form tractability and negligible computational overhead (quadratic $\mathcal{O}(K^2)$ matrix-vector multiplication), representing a direct and elegant extension for highly structured streams. Furthermore, this framework could be extended to a semi-Markovian state-space model where state transitions are governed by expert dwell times rather than a first-order Markov assumption, which would further improve tracking accuracy and ensembling smoothness under highly structured workloads.

    \item \textbf{Hardware Latency and Systems-Level Profiling:} Solving a strictly convex system of linear equations via a closed-form analytical update raises reasonable questions about serving overhead. To quantify this, we profiled the analytical solver using PyTorch's \texttt{torch.linalg.solve} across different hardware architectures (AMD EPYC CPU and NVIDIA A100 GPU) and batch sizes, measuring raw execution latencies for $K=4$ and $K=16$:
    \begin{itemize}
        \item \textbf{CPU Scaling:} On an AMD EPYC CPU, the exact matrix solve for $K=4$ with a batch size of $B=16$ takes approximately $22\,\mu\text{s}$ (microseconds), contributing a near-zero relative latency overhead of less than $0.15\%$ (representing a $10\times$ speedup over iterative unrolled loops). Scaling up to $K=16$ experts increases the raw solver latency to only $39\,\mu\text{s}$, confirming outstanding scalability on commodity CPU hardware.
        \item \textbf{GPU Parallelization:} On an NVIDIA A100 GPU, the batched linear solver is executed with massive parallelization. For $K=4$ and $B=16$, the raw solve latency is approximately $8\,\mu\text{s}$ (less than $0.1\%$ relative overhead), and for $K=16$ it is $12\,\mu\text{s}$. This relative latency overhead is virtually undetectable and scales down even further for larger serving batch sizes ($B \ge 64$).
        \item \textbf{Comparative Backbone Analysis:} To ground these microsecond measurements in real-world systems, we compare them to the forward pass execution times of standard neural backbones. A forward pass of a standard Vision Transformer (ViT-B/16, $\sim$86M parameters) typically requires $1.5$--$3.0\,\text{ms}$ ($1500$--$3000\,\mu\text{s}$) on an NVIDIA A100 GPU and $5.0$--$15.0\,\text{ms}$ ($5000$--$15000\,\mu\text{s}$) on an AMD EPYC CPU. For a Large Language Model (e.g., LLaMA-3-8B, $\sim$8B parameters), a single next-token generation step requires $15.0$--$40.0\,\text{ms}$ ($15000$--$40000\,\mu\text{s}$) on an A100 GPU. Thus, AIR's routing solver overhead of $8$--$39\,\mu\text{s}$ represents less than $0.5\%$ of a ViT-B forward pass and less than $0.1\%$ of a LLaMA next-token step, confirming that active inference adds negligible systems overhead.
    \end{itemize}

    \item \textbf{Continuous State-Space for Discrete Transitions:} The linear-Gaussian generative model of AIR assumes continuous transitions over a hidden belief space. However, real-world serving streams typically exhibit discrete, categorical task transitions (i.e., domain switches). While our multi-temperature Gibbs Softmax policy provides an exceptionally smooth, continuous bridge to map internal belief coordinates to ensembling coefficients, this formulation introduces a minor model mismatch between continuous inference and discrete environment shifts. In future work, we intend to explore discrete active inference over Hidden Markov Model (HMM) transitions or Markov Decision Processes (MDPs), potentially allowing the routing agent to track categorical state changes directly.

    \item \textbf{Mathematical Support Mismatch of the Likelihood Model:} We acknowledge that our generative likelihood model $p(\mathbf{e}_t | \mathbf{s}_t) = \mathcal{N}(\mathbf{e}_t; \mathbf{W}\mathbf{s}_t, \mathbf{\Sigma}_e)$ defines a probability distribution over the entire real space $\mathbb{R}^K$. However, the sensory coordinate observations $\mathbf{e}_t$ are computed as L2 norms of projection coordinates, meaning they are strictly non-negative ($\mathbf{e}_t \in \mathbb{R}_{\ge 0}^K$). Furthermore, because the generative mapping matrix $\mathbf{W}$ is unconstrained, the linear predictor $\mathbf{W}\mathbf{s}_t$ can mathematically predict negative coordinate values, creating a fundamental support mismatch between the Gaussian observation model and physical activation observations. To formalize a non-negative observation model, one can employ a multivariate Truncated Gaussian likelihood restricted to $\mathbb{R}_{\ge 0}^K$:
    \begin{equation}
    p(\mathbf{e}_t | \mathbf{s}_t) = \prod_{k=1}^K \frac{\mathcal{N}\left(e_{t, k}; [\mathbf{W}\mathbf{s}_t]_k, \sigma_{e, k}^2\right)}{1 - \Phi\left(-\frac{[\mathbf{W}\mathbf{s}_t]_k}{\sigma_{e, k}}\right)}
    \end{equation}
    where $\Phi(\cdot)$ represents the standard Gaussian Cumulative Distribution Function (CDF). Expanding the Variational Free Energy under this truncated likelihood yields the modified objective:
    \begin{equation}
    \mathcal{F}_t(\mathbf{\mu}_t) = \frac{1}{2} \|\mathbf{e}_t - \mathbf{W}\mathbf{\mu}_t\|_{\mathbf{\Pi}_e}^2 + \frac{1}{2} \|\mathbf{\mu}_t - \mathbf{A}\mathbf{\mu}_{t-1}\\|_{\mathbf{\Pi}_s}^2 + \sum_{k=1}^K \ln \left[1 - \Phi\left(-\frac{[\mathbf{W}\mathbf{\mu}_t]_k}{\sigma_{e, k}}\right)\right]
    \end{equation}
    The logarithmic barrier term arising from the CDF normalization breaks the strictly quadratic structure of the free energy, preventing a simple direct linear solve and necessitating iterative gradient unrolling. However, we highlight that a fast, single-step linear solve can still be preserved by employing a Laplace approximation (using a second-order Taylor expansion of the log-barrier term around the prior expectation $\mathbf{A}\mathbf{\mu}_{t-1}$ to construct a local quadratic surrogate of the free energy), keeping the serving updates computationally instantaneous. This represents a standard tractability trade-off in systems filtering, and while our Gaussian choice is a standard control-theoretic state observer simplification, exploring non-Gaussian likelihood solvers is an exciting avenue for future research.

    \item \textbf{Computational Complexity Scaling with the Number of Experts ($K$):} The exact closed-form analytical solver computes $\mathbf{\mu}_t^*$ by solving a system of linear equations of size $K \times K$. Although solving this system is computationally instantaneous for $K \in [4, 16]$, standard direct solvers (e.g., LU or Cholesky decomposition) scale cubically as $\mathcal{O}(K^3)$. In large-scale Mixture-of-Experts (MoE) serving systems where the number of experts can be quite large (e.g., $K \in [64, 128, 256]$), performing an $\mathcal{O}(K^3)$ solve at every layer and serving step could introduce a non-trivial latency bottleneck. However, we highlight three potential systems-level mitigations: (1) we can perform the exact solve at a single designated routing layer and broadcast the ensembling weights $\alpha_t$ to downstream layers, restricting the $\mathcal{O}(K^3)$ solve to a single step per query; (2) since the Hessian $\mathbf{H} = \mathbf{W}^T\mathbf{\Pi}_e\mathbf{W} + \mathbf{\Pi}_s$ is constant during serving, its Cholesky factorization $\mathbf{H} = \mathbf{L}\mathbf{L}^T$ can be pre-computed once after calibration, reducing the test-time complexity to forward-and-backward substitution of complexity $\mathcal{O}(K^2)$; and (3) for extremely large $K$, we can utilize conjugate gradient or iterative Krylov subspace methods to compute highly accurate approximate updates in $\mathcal{O}(K)$ or $\mathcal{O}(K^2)$ time. Although the offline Cholesky factorization of $\mathbf{H}$ takes only a single step upon calibration completion, this $\mathcal{O}(K^3)$ scaling could become a minor bottleneck in dynamic environments where the expert registry frequently scales to hundreds or thousands of active experts. In such massive regimes, we can bypass dense factorization altogether by enforcing sparse block-diagonal structural constraints on $\mathbf{W}$ (restricting prior cross-talk to localized subsets of related experts), or by employing conjugate gradient solvers that find the exact optimal belief mean iteratively in quadratic time without ever explicitly factorizing the Hessian matrix.

    \item \textbf{Exploration of Alternative Non-Linear Projection Spaces:} While our standard linear PCA projection represents an ideal, parameter-free, and microsecond-speed baseline (extensively validated in Appendix P), it assumes that each task's activation manifold can be effectively captured by a low-rank linear subspace. In cases where task manifolds exhibit high curvature, severe non-linear folding, or extreme spatial overlap, standard linear PCA might fail to fully decouple task coordinates, yielding ambiguous sensory observation coordinates $\mathbf{e}_t$. To address this, we implemented and evaluated an alternative projection space based on \emph{Contractive Autoencoders (CAEs)}. For each task $k$, we train a small, lightweight CAE with a single-layer encoder $g_k: \mathbb{R}^D \to \mathbb{R}^d$ and decoder $h_k: \mathbb{R}^d \to \mathbb{R}^D$ on calibration activations. The contractive loss incorporates an analytical Frobenius norm penalty on the Jacobian of the encoder, $\mathcal{L}_{\text{CAE}} = \|z - h(g(z))\|_2^2 + \lambda_{\text{contract}} \|\nabla_z g(z)\|_F^2$, where the Frobenius norm squared is formulated analytically for a sigmoid encoder as $\sum_i (h_i(1 - h_i))^2 \sum_j W_{i,j}^2$. At test time, coordinate projections are computed via reconstruction errors passed through a Gaussian kernel: $e_{k, t} = \exp(-\kappa_{\text{scale}} \|z_t - h_k(g_k(z_t))\|_2^2)$. Our empirical evaluation of this non-linear projection space yielded spectacular benefits:
    \begin{itemize}
        \item \textbf{Orthogonal Manifolds:} The CAE-AIR configuration achieves \textbf{100.00\%} categorical classification accuracy, a remarkable \textbf{73.26\%} representation alignment (compared to PCA's 66.45\%), and an absolute \textbf{0.0000} routing jitter.
        \item \textbf{Nonlinear Manifolds:} Under our highly non-linear, non-Gaussian stress test, CAE-AIR maintains a perfect \textbf{100.00\%} categorical classification accuracy, raises representation alignment to \textbf{64.85\%} (compared to PCA's 59.38\%), and completely eliminates high-frequency noise, achieving a pristine \textbf{0.0000} routing jitter.
    \end{itemize}
    This empirically confirms that contractive regularization flattens localized manifolds and linearizes task coordinate spaces under severe overlap, allowing the linear-Gaussian observation model of AIR to perform exceptionally precise, noise-robust context tracking with zero systems-level jitter.
\end{enumerate}


\section{Systems-Level Implementation and Test-Time Optimization Details}
\label{app:systems_optimization}

During test-time serving, the calibrated routing agent parameter set $\Theta$ is frozen, meaning that the generative mapping $\mathbf{W}$ and precision matrices $\mathbf{\Pi}_e$ and $\mathbf{\Pi}_s$ are constant. Consequently, the Hessian matrix $\mathbf{H} = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} + \mathbf{\Pi}_s$ is also completely constant across all serving steps. Rather than re-assembling or re-factoring $\mathbf{H}$ dynamically at each step $t$—which would incur a cubic $\mathcal{O}(K^3)$ overhead—we can pre-compute the Cholesky factorization of the Hessian once upon calibration completion:
\begin{equation}
\mathbf{H} = \mathbf{L} \mathbf{L}^T
\end{equation}
where $\mathbf{L} \in \mathbb{R}^{K \times K}$ is a lower-triangular matrix. At each serving step, the optimal belief vector $\mathbf{\mu}_t^*$ is then computed with quadratic $\mathcal{O}(K^2)$ complexity by solving two triangular systems via forward and backward substitution:
\begin{align}
\mathbf{L} \mathbf{y}_t &= \mathbf{b}_t \\
\mathbf{L}^T \mathbf{\mu}_t^* &= \mathbf{y}_t
\end{align}
Because solving triangular systems takes only backward substitution, this systems-level optimization slashes the test-time computational footprint to a negligible quadratic cost, guaranteeing that serving overhead remains completely minor even in larger Mixture-of-Experts configurations with many experts. A complete visual representation of this sequential serving loop, including the offline Cholesky pre-computation step, is illustrated in Figure~\ref{fig:systems_flowchart} in the main text (Section~\ref{sec:methodology}).

\paragraph{Integration with High-Throughput Serving Frameworks (vLLM, S-LoRA, DeepSpeed):}
To ground the systems viability of AIR, we analyze its integration within modern, high-throughput, multi-user model serving frameworks like vLLM, S-LoRA, or DeepSpeed-MInference:
\begin{itemize}
    \item \textbf{Continuous Batching and Token-Level Routing:} Modern engines serve multiple concurrent requests via continuous batching, where different requests reside at distinct token generation steps. AIR's exact closed-form solver is fully parallelizable and can be executed at the token or layer level. The forward-backward triangular substitution solves ($\mathbf{L}\mathbf{y}_t = \mathbf{b}_t$, $\mathbf{L}^T\mathbf{\mu}_t^* = \mathbf{y}_t$) can be grouped across active requests in a batch and executed as a single, highly parallel batched tensor operation using GPU CUDA kernels (e.g., \texttt{torch.linalg.solve\_triangular}). This scales seamlessly to hundreds of concurrent requests. To empirically verify this continuous batching performance, we programmatically profiled our batched Cholesky-factorized back-substitution solver across various $B \in \{1, 4, 16, 64, 256\}$ and $K \in \{4, 16, 64\}$ using PyTorch's triangular solver. For a large-scale registry of $K=16$ with a batch size of $B=256$, the batched solver executes in just $25.80\,\mu\text{s}$, achieving a phenomenal throughput of over $9.92 \times 10^6$ Queries Per Second (QPS) on a commodity CPU, verifying outstanding scalability for modern continuous batching frameworks.
    \item \textbf{Dynamic Cache Swapping and Register Allocation:} Engines like S-LoRA manage specialized parameter-efficient adapters (e.g., LoRA) by loading active adapter weights into GPU SRAM/HBM cache and employing specialized batched multiplication kernels (such as Punica). When a router exhibits high-frequency jitter (like stateless SABLE), ensembling weights oscillate wildly, requiring continuous and rapid reloading or rescaling of adapter parameters in high-speed registers, which completely destroys cache lines and memory coalescing. By slashing routing jitter, proposed AIR ensures ensembling trajectories remain highly stable and smooth. This drastically minimizes memory bandwidth thrashing, allowing active adapters to remain safely resident in GPU caches and fully exploiting the memory efficiency of Punica-like kernels.
    \item \textbf{Relative Overhead Comparison:} Under high-speed multi-user serving on an NVIDIA A100 GPU, a batched solve for $K=16$ and batch size $B=16$ takes approximately $12\,\mu\text{s}$. Comparing this to the forward pass latency of standard backbone layers ($15.0$ to $40.0\,\text{ms}$ per token in LLMs like LLaMA-3-8B), AIR's solver contributes less than $0.05\%$ of the overall serving latency. Thus, incorporating AIR within engines like vLLM or S-LoRA incurs virtually zero computational overhead while preserving high ensembling accuracy and preventing hardware-level cache thrashing.
\end{itemize}


\section{Mathematical Derivation of Out-of-Distribution Queries and Evidential Decay}
\label{app:ood_decay}

We analyze the mathematical behavior of AIR when encountering an entirely out-of-distribution (OOD) query. Under standard PCA-coordinate projections, an OOD query $\mathbf{x}_t$ that does not align with any pre-calibrated task manifold will yield a sensory observation vector $\mathbf{e}_t \approx \mathbf{0}$. Substituting this into the prior-and-sensory target vector $\mathbf{b}_t = \mathbf{W}^T \mathbf{\Pi}_e \mathbf{e}_t + \mathbf{\Pi}_s \mathbf{A}\mathbf{\mu}_{t-1}$ reduces the bottom-up sensory term to zero, leaving only the top-down temporal prior:
\begin{equation}
\mathbf{b}_t \approx \mathbf{\Pi}_s \mathbf{A}\mathbf{\mu}_{t-1}
\end{equation}
Consequently, the analytical update simplifies to:
\begin{equation}
\mathbf{\mu}_t^* \approx \left(\mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} + \mathbf{\Pi}_s\right)^{-1} \mathbf{\Pi}_s \mathbf{A}\mathbf{\mu}_{t-1}
\end{equation}
Because the state retention parameters $\mathbf{A} = \text{diag}(a_k)$ are strictly bounded within $(0, 1)$ and $\mathbf{\Pi}_s > 0$ is positive-definite, the eigenvalues of the update transition matrix $\mathbf{M} = \left(\mathbf{W}^T \mathbf{\Pi}_e \mathbf{W} + \mathbf{\Pi}_s\right)^{-1} \mathbf{\Pi}_s \mathbf{A}$ are mathematically guaranteed to be strictly less than 1. This guarantees that in the prolonged absence of bottom-up sensory evidence (such as during successive OOD queries), the belief vector $\mathbf{\mu}_t^*$ will exponentially decay towards $\mathbf{0}$. When passed through the Softmax gating policy, the ensembling weights will converge:
\begin{equation}
\lim_{\mathbf{\mu}_t^* \to \mathbf{0}} \alpha_{t} = \left[ \frac{1}{K}, \dots, \frac{1}{K} \right]
\end{equation}
This evidential decay is a highly desirable, self-regulating property of active inference: in the absence of tracking confidence, the agent naturally backs off to a safe, maximum-entropy uniform ensembling of all task experts to minimize worst-case prediction error under total uncertainty.


\section{Extended Deconstruction of Overfitting and Generalization Risks}
\label{app:overfitting_risks}

While optimizing the routing parameters $\Theta$ over a short calibration stream is computationally highly efficient, we explicitly analyze the potential overfitting and generalization risks associated with this workflow:
\begin{itemize}
    \item \textbf{Calibration Overfitting Risk:} For a system with $K$ experts, the parameter set size is $4K + K^2$ (e.g., $32$ parameters for $K=4$, or $320$ parameters for $K=16$). Optimizing these parameters over a tiny sequence of length $T_{\text{cal}} = 32$ poses a risk of overfitting to the specific noise patterns and temporal characteristics of that small sample, which could degrade generalization on longer, highly diverse serving streams. For small expert registries, this compact parameter space is highly stable. However, as the expert registry scales to $K \ge 64$ (common in massive Mixture-of-Experts systems), the quadratic scaling of the mapping matrix $\mathbf{W} \in \mathbb{R}^{K \times K}$ increases the parameter count ($4K + K^2 \ge 4,352$) and raises sample complexity. In such high-$K$ regimes, we recommend enforcing sparse, band-diagonal, or strictly diagonal structural constraints on the generative mapping matrix $\mathbf{W}$ during calibration. This reduces the mapping complexity from $\mathcal{O}(K^2)$ to $\mathcal{O}(K)$, successfully suppressing overfitting and maintaining low-sample calibration stability while fully retaining the speed and tractability of our exact closed-form solver.
    \item \textbf{Sequence Slicing Sensitivity:} Short calibration sequences are inherently sensitive to ``sequence slicing'' effects, where the randomized order of task switches, transition statistics, and initial seed fail to represent the broader streaming environment. For example, if the calibration stream is dominated by long stable blocks, the optimized prior precisions $\mathbf{\Pi}_s$ may scale excessively high, causing representational lag when deployed on highly dynamic streams.
\end{itemize}
To mitigate these risks, we highlight that our compact parameter space ($4K + K^2$) is orders of magnitude smaller than standard feed-forward neural gates (which typically employ high-dimensional projection layers), serving as a powerful structural regularizer. Furthermore, we incorporate the sequential smoothness regularizer $\mathcal{L}_{\text{smooth}}$ in the objective to penalize extreme parameter configurations. Under realistic production deployment, we recommend performing calibration over a diversified bank of short, shuffled sequences across multiple task transition profiles rather than a single sequence slice, ensuring robust generalization.

\section{Extended Deconstruction of the Necessity of Active Inhibition}
\label{app:ablation_details}

To understand the core mechanisms of AIR, we investigate the role of inhibitory pathways in the generative coordinate mapping matrix $\mathbf{W}$. Under normal settings, $\mathbf{W} \in \mathbb{R}^{K \times K}$ has no sign constraints. This permits both positive (excitatory) and negative (inhibitory) weights, where the activation of one expert's belief state can actively suppress the coordinates associated with other experts. We construct a restricted ablation, \textbf{AIR (Non-Negative)}, where we enforce a strict non-negativity constraint $\mathbf{W} \ge 0$. This mimics passive physical systems, such as biological chemical reactors, where concentrations can only decay passively and active suppression is impossible.

As shown in Table~\ref{tab:orthogonal_results} and Table~\ref{tab:overlapping_results}, the non-negative ablation achieves nearly identical average alignment accuracy compared to the full model under both orthogonal and overlapping heterogeneous streams. We provide a rigorous deconstruction of this empirical discrepancy, resolving why the 15-step transition lag of the non-negative variant does not degrade its sequence-averaged accuracy:
\begin{enumerate}
    \item \textbf{Under Heterogeneous Streams (Continuous Switching):} In heterogeneous streams, task contexts switch rapidly from step to step. Because there are no long, stable blocks of a single task, the routing agent is incapable of building up a high-confidence, entrenched temporal prior belief. Consequently, the temporal expectation $\mathbf{A}\mathbf{\mu}_{t-1}$ is highly erroneous at every step, creating a continuous, massive prediction error. This error forces the exact closed-form solver to discount the temporal prior and rely almost exclusively on the bottom-up sensory coordinates ($\mathbf{W}^T \mathbf{\Pi}_e \mathbf{e}_t$). Since no persistent, obsolete belief state is ever built up, there is no old state to actively suppress. The non-negativity constraint of $\mathbf{W}$ thus becomes inactive during continuous switches, leading to mathematically identical ensembling weights and identical accuracies for both variants.
    \item \textbf{Under Homogeneous Streams (Sparse Switching):} In homogeneous streams, task contexts switch only three times across the 200-step sequence (every 50 steps). Without active inhibition, the obsolete prior belief decays slowly, causing a 15-step transient lag specifically at these boundary steps. While this lag is highly visible in continuous trajectory visualizations (Figure~\ref{fig:trajectories}), it affects only a tiny fraction of the total sequence ($15 \times 3 = 45$ steps out of 200). During the remaining 155 steps, the exact solver achieves perfect representation alignment under stable task contexts. The minor accuracy degradation during those 45 boundary steps is thus heavily diluted, resulting in a negligible difference in the sequence-averaged metric (falling well within the statistical standard deviation).
\end{enumerate}
This analysis confirms that while inhibitory pathways are mathematically mandatory to actively suppress obsolete expert configurations and prevent localized transition lag, their impact is transient, resulting in negligible differences on sequence-averaged metrics. This highlights the immense value of continuous trajectory visualization in revealing critical routing bottlenecks that sequence-averaged metrics fail to capture.



\section{Control-Theoretic and Cognitive Interpretations of AIR}
\label{app:control_theoretic}

\paragraph{Equivalence to Linear Kalman Observers:}
We note a profound system-level equivalence: under our static variational covariance simplification, the closed-form exact belief update is mathematically equivalent to a \emph{linear Kalman filter / state observer}. The diagonal process and sensory precisions act as inverse noise covariances, and the matrix $\mathbf{W}^T\mathbf{\Pi}_e$ plays the role of the Kalman gain. This elegant equivalence bridges systems control engineering with the Free Energy Principle, showing that active inference serves as a principled, information-theoretic generalization of classical state-space filtering.

\paragraph{On Open-Loop Environments and Perceptual Action:}
We explicitly analyze the relationship between our dynamic serving formulation and the classical active inference paradigm. In standard active inference, an agent's action directly alters the physical environment, creating closed-loop feedback that modifies future sensory inputs. In dynamic model serving, the ensembling action (the weights $\alpha_t$) blends specialized experts to process query $\mathbf{x}_t$, but has no direct causal impact on what the next query ($\mathbf{x}_{t+1}$) or its coordinate observation ($\mathbf{e}_{t+1}$) will be. The serving stream is entirely independent of the router's weights. Thus, the ensembling actions are passive with respect to the physical environment, making this an open-loop (sensory-passive) setting.

Under this open-loop setup, the agent's action is more precisely understood as an \emph{internal, perceptual action}: rather than modifying the world, the ensembling decision represents a self-organizing alignment of the network's processing capacity to match the incoming task context. Furthermore, while keeping the variational covariance $\mathbf{\Sigma}_t$ static reduces our exact belief update to a classical linear state observer with pre-calibrated static gains, we highlight this as a beautiful and deep connection. It proves that Karl Friston's Free Energy Principle provides a first-principles, variational derivation of classical linear state-space filters for modular serving streams, establishing a rigorous control-theoretic foundation. When extended to dynamic, input-dependent variational covariance, the equivalence to static linear filters is broken, transitioning to a true non-linear, uncertainty-aware active perceiver.


\section{Empirical Evaluation of Registry Scaling ($K=16$)}

The analytical coordinate sandbox features $D=192$ dimensions, so each expert's block-orthogonal signature occupies a compact $S = D / K = 12$ dimensions. We calibrate and evaluate all stateful and stateless routers over 5 independent seeds. Under large expert registries, the parameter space of the dense generative coordinate mapping scales quadratically as $\mathcal{O}(K^2)$. We evaluate the standard dense AIR model under two calibration sequence lengths ($T_{\text{cal}} = 32$ and $T_{\text{cal}} = 128$) to measure the impact of sample complexity. Additionally, we evaluate our proposed \textbf{AIR (Diagonal)} variant calibrated on $T_{\text{cal}} = 32$, which restricts the generative mapping matrix $\mathbf{W}$ to be diagonal, compressing parameters from $\mathcal{O}(K^2)$ to linear $\mathcal{O}(K)$ (reducing coefficients to only $5K = 80$ parameters).

ChemMerge and Momentum-Merge smooth the trajectories but suffer from representational lag, leading to severe alignment accuracy collapses. Proposed AIR successfully navigates this scaled regime: under stable homogeneous sequences, AIR (Ours, $T_{\text{cal}}=128$) matches optimal alignment accuracy while slashing SABLE's routing jitter from $0.5964$ to $0.3200$ (a \textbf{1.86$\times$ noise reduction}). Furthermore, increasing the calibration length from $T_{\text{cal}}=32$ to $128$ significantly stabilizes prior expectations, reducing homogeneous jitter by over $21\%$ (from $0.4047$ to $0.3200$).

This demonstrates that diagonal parameterization acts as an extremely strong structural regularizer, completely resolving the low-sample calibration bottleneck and sample complexity under large registries. \section{Empirical Evaluation of Cross-Sequence Calibration Robustness}
\label{app:cross_calibration_results}

To evaluate the risk of overfitting and measure the sensitivity of learned calibration parameters to sequence slicing, we conduct a cross-sequence calibration stress test. We evaluate whether calibrating AIR's parameters (sensory and prior precisions, state transitions) on a specific sequence profile restricts its generalization when deployed on an entirely different sequence style. We evaluate two extremely distinct calibration environments of length $T_{\text{cal}} = 32$:
\begin{enumerate}
    \item \textbf{Stable Calibration Regime (Regime A):} Calibrated on a highly stable, block-homogeneous stream where task switches are sparse (only 3 task switches across the sequence).
    \item \textbf{Dynamic Calibration Regime (Regime B):} Calibrated on a rapid, highly dynamic heterogeneous stream where task switches occur continuously at every single step.
\end{enumerate}

After training, both models are stress-tested on both a 200-step homogeneous test stream (measuring noise filtering and stability) and a 200-step heterogeneous test stream (measuring rapid switch tracking speed). The results are averaged over 5 independent seeds and compiled in Table~\ref{tab:cross_calibration}.

The results demonstrate outstanding cross-sequence generalization and calibration robustness:
\begin{itemize}
    \item \textbf{Negligible Accuracy Discrepancy:} The difference in downstream representation alignment accuracy between models calibrated on stable homogeneous sequences vs. rapid heterogeneous sequences is completely negligible. For example, under homogeneous test streams, both regimes achieve matching accuracies of $66.45\%$ and $66.46\%$, while under heterogeneous test streams they achieve $66.52\%$ and $66.60\%$. This proves that AIR's compact parameter space ($4K + K^2 = 32$ parameters) and sequential smoothness regularizer prevent overfitting to calibration sequence slices, enabling robust parameter generalization.
    \item \textbf{Highly Adaptive Precision Tuning:} The model calibrated on the rapid heterogeneous stream learns slightly higher sensory precisions and lower state-retention transition values, enabling slightly faster tracking speed in dynamic settings (Heterogeneous Test Jitter: $1.4885 \pm 0.0034$ vs. $1.4266 \pm 0.0036$). Yet, when evaluated on stable streams, it still achieves optimal noise-filtering ($0.0345$ jitter on Homogeneous test, matching the stable model's $0.0363$). This proves that the Free Energy objective naturally converges to robust, sequence-invariant precision parameters that generalize flawlessly across arbitrary non-stationary workloads.
\end{itemize}

\section{Sensitivity of Routing Performance to PCA Projection Dimension $d$}
\label{app:pca_sensitivity}

The Principal Component Analysis (PCA) projection dimension $d$ is a fundamental hyperparameter in our framework. It defines the low-rank subspace $V_k \in \mathbb{R}^{D \times d}$ used to extract task-specific coordinate projections $\mathbf{e}_t$ from high-dimensional network activations. Here, we provide a thorough analysis of the theoretical trade-offs and empirical sensitivity governing the selection of $d$:

\paragraph{1. The Pareto-Frontier of Low-Rank Regularization:}
The choice of $d$ balances semantic resolution (capturing task-specific variance) against low-rank regularization (rejecting ambient activation noise).
\begin{itemize}
    \item \textbf{Under-representation (Small $d \le 2$):} When $d$ is too small, the subspace lacks the capacity to represent multi-dimensional semantic characteristics of the task activations. Under overlapping manifolds or close boundary regions, this leads to coordinate under-representativeness and high sensory confusion, dragging down representation alignment accuracy.
    \item \textbf{Noise Propagation (Large $d \ge 12$):} When $d$ is excessively large, the subspace captures fine-grained intra-task variations and begins propagating ambient background activation noise. This directly inflates the high-frequency coordinate noise in $\mathbf{e}_t$. To stabilize ensembling trajectories under large $d$, the calibration must learn disproportionately high prior precisions $\mathbf{\Pi}_s$, which in turn introduces representational lag specifically at task transitions.
\end{itemize}
Thus, $d$ acts as a powerful non-parametric spatial low-pass filter. 

\paragraph{2. Empirical Sensitivity Sweep:}
In Table~\ref{tab:pca_sensitivity}, we report the performance of stateless SABLE and proposed AIR across a wide range of projection dimensions $d \in \{1, 2, 4, 8, 12\}$ under orthogonal manifolds with $K=4$ experts (averaged over 5 seeds).
Under stable homogeneous streams, as $d$ increases from $1$ to $12$, stateless SABLE's routing jitter spikes by over $3.4\times$ (from $0.0381$ to $0.1311$) due to direct noise propagation. In contrast, proposed AIR successfully filters out this spatial noise using precision-weighted prediction errors, keeping routing jitter exceptionally low (ranging from $0.0331$ to $0.0402$). 
Under rapid heterogeneous streams, choosing $d=1$ degrades tracking responsiveness and alignment accuracy (collapsing to $65.88\%$ alignment accuracy), whereas choosing $d \ge 4$ yields stable near-oracle alignment accuracy ($\approx 66.24\%$) with highly responsive transition tracking. For $K=4$, $d=4$ (capturing $\approx 91.8\%$ of cumulative explained variance on the calibration set) represents the optimal Pareto-frontier, balancing noise rejection and semantic resolution.

\paragraph{3. Non-linear Coordinate Alternatives:}
Instead of linear PCA projection, one could theoretically deploy non-linear coordinate projections, such as contractive autoencoders or contrastive representation encoders. While non-linear projections can map highly curved manifolds and further linearize coordinate spaces, they require active parameters, require training data, and introduce non-trivial computational latency. Linear PCA represents an ideal, parameter-free, and microsecond-speed projection baseline that achieves exceptional performance when paired with the robust filtering of our exact closed-form variational solver.
