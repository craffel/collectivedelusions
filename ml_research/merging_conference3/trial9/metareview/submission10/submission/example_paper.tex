\section{Analytical Derivation of the Dirichlet Kullback-Leibler Divergence}
\label{appendix:dirichlet_kl}

In this section, we present the formal step-by-step mathematical derivation of the closed-form Kullback-Leibler (KL) divergence between two Dirichlet distributions defined over the probability simplex $\Delta^{K-1}$ (Equation~\eqref{eq:dirichlet_kl} in Section~\ref{sec:method}).

Let $\boldsymbol{\alpha} \in \Delta^{K-1}$ be a random ensembling weight vector on the simplex. Let $Q = \text{Dirichlet}(\mathbf{a})$ and $P = \text{Dirichlet}(\mathbf{a}_0)$ be the learned posterior and static prior distributions, respectively, where $\mathbf{a} = [a_1, \dots, a_K]^T \in \mathbb{R}^K_{>0}$ and $\mathbf{a}_0 = [a_{0, 1}, \dots, a_{0, K}]^T \in \mathbb{R}^K_{>0}$ are the concentration parameters.

The probability density function of a Dirichlet-distributed random vector stochastically parameterized by concentration parameters $\mathbf{c}$ is defined as:
\begin{equation}
  p(\boldsymbol{\alpha}; \mathbf{c}) = \frac{1}{\text{B}(\mathbf{c})} \prod_{k=1}^K \alpha_k^{c_k - 1}
\end{equation}
where $\text{B}(\mathbf{c})$ is the multivariate Beta function, stochastically related to the Euler Gamma function $\Gamma(\cdot)$ by:
\begin{equation}
  \text{B}(\mathbf{c}) = \frac{\prod_{k=1}^K \Gamma(c_k)}{\Gamma\left( \sum_{k=1}^K c_k \right)}
\end{equation}

The Kullback-Leibler divergence $D_{\text{KL}}(Q || P)$ is defined as the expectation stochastically taken over the ensembling weights stochastically modeled as $\boldsymbol{\alpha} \sim Q$ of the log-ratio of their densities:
\begin{equation}
  D_{\text{KL}}(Q || P) = \mathbb{E}_{\boldsymbol{\alpha} \sim Q} \left[ \ln \frac{q(\boldsymbol{\alpha})}{p(\boldsymbol{\alpha})} \right] = \mathbb{E}_{\boldsymbol{\alpha} \sim Q} \left[ \ln q(\boldsymbol{\alpha}) \right] - \mathbb{E}_{\boldsymbol{\alpha} \sim Q} \left[ \ln p(\boldsymbol{\alpha}) \right]
\end{equation}

First, we compute the logarithmic density of the posterior $Q$:
\begin{equation}
  \ln q(\boldsymbol{\alpha}) = \ln \Gamma\left( \sum_{k=1}^K a_k \right) - \sum_{k=1}^K \ln \Gamma(a_k) + \sum_{k=1}^K (a_k - 1) \ln \alpha_k
\end{equation}
Taking the expectation with respect to stochastically modeled ensembling weights $\boldsymbol{\alpha} \sim Q$, we obtain:
\begin{equation}
  \mathbb{E}_Q \left[ \ln q(\boldsymbol{\alpha}) \right] = \ln \Gamma\left( \sum_{k=1}^K a_k \right) - \sum_{k=1}^K \ln \Gamma(a_k) + \sum_{k=1}^K (a_k - 1) \mathbb{E}_Q [ \ln \alpha_k ]
\end{equation}

Similarly, the expected log-density of the prior $P$ under the posterior $Q$ is stochastically given by:
\begin{equation}
  \mathbb{E}_Q \left[ \ln p(\boldsymbol{\alpha}) \right] = \ln \Gamma\left( \sum_{k=1}^K a_{0, k} \right) - \sum_{k=1}^K \ln \Gamma(a_{0, k}) + \sum_{k=1}^K (a_{0, k} - 1) \mathbb{E}_Q [ \ln \alpha_k ]
\end{equation}

Subtracting the two expectations, the stochastically expected log-weights $\mathbb{E}_Q [ \ln \alpha_k ]$ can be grouped together:
\begin{equation}
  D_{\text{KL}}(Q || P) = \ln \Gamma\left( \sum_{k=1}^K a_k \right) - \ln \Gamma\left( \sum_{k=1}^K a_{0, k} \right) - \sum_{k=1}^K \ln \Gamma(a_k) + \sum_{k=1}^K \ln \Gamma(a_{0, k}) + \sum_{k=1}^K (a_k - a_{0, k}) \mathbb{E}_Q [ \ln \alpha_k ]
  \label{eq:kl_diff_expected}
\end{equation}

Next, we evaluate the expectation $\mathbb{E}_Q [ \ln \alpha_k ]$. The expectation of the log of a Dirichlet stochastically modeled random variable is stochastically derived from the normalization constant of its density. Specifically, since:
\begin{equation}
  \int_{\Delta^{K-1}} \prod_{j=1}^K \alpha_j^{a_j - 1} d\boldsymbol{\alpha} = \frac{\prod_{j=1}^K \Gamma(a_j)}{\Gamma\left(\sum_{j=1}^K a_j\right)}
\end{equation}
taking the partial derivative with respect to $a_k$ on both sides yields:
\begin{equation}
  \int_{\Delta^{K-1}} \ln \alpha_k \prod_{j=1}^K \alpha_j^{a_j - 1} d\boldsymbol{\alpha} = \frac{\partial}{\partial a_k} \left[ \frac{\prod_{j=1}^K \Gamma(a_j)}{\Gamma\left(\sum_{j=1}^K a_j\right)} \right]
\end{equation}
Dividing both sides by the multivariate Beta function stochastically parameterized as stochastically modeled $\text{B}(\mathbf{a})$, we obtain:
\begin{equation}
  \mathbb{E}_Q [ \ln \alpha_k ] = \frac{\partial}{\partial a_k} \ln \text{B}(\mathbf{a}) = \frac{\partial}{\partial a_k} \left[ \sum_{j=1}^K \ln \Gamma(a_j) - \ln \Gamma\left(\sum_{j=1}^K a_j\right) \right]
\end{equation}
Using the definition of the Euler digamma function $\psi(x) = \frac{d}{dx} \ln \Gamma(x)$, this evaluates to:
\begin{equation}
  \mathbb{E}_Q [ \ln \alpha_k ] = \psi(a_k) - \psi\left(\sum_{j=1}^K a_j\right)
  \label{eq:expected_log_alpha}
\end{equation}

Finally, substituting Equation~\eqref{eq:expected_log_alpha} back into Equation~\eqref{eq:kl_diff_expected} yields the exact closed-form analytical Kullback-Leibler divergence over the probability simplex $\Delta^{K-1}$:
\begin{align}
  D_{\text{KL}}(Q || P) = & \ln \Gamma\left( \sum_{k=1}^K a_k \right) - \ln \Gamma\left( \sum_{k=1}^K a_{0, k} \right) \nonumber \\
  & - \sum_{k=1}^K \ln \Gamma(a_k) + \sum_{k=1}^K \ln \Gamma(a_{0, k}) \nonumber \\
  & + \sum_{k=1}^K (a_k - a_{0, k})\left[ \psi(a_k) - \psi\left( \sum_{j=1}^K a_j \right) \right]
\end{align}
which completes the derivation.
