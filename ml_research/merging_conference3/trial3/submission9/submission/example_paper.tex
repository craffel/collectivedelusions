\section{The FlatQ-Merge Algorithm}
\label{sec:appendix_algorithm}

We provide the detailed step-by-step procedural flow of FlatQ-Merge in Algorithm \ref{alg:flatq_merge}.

\begin{algorithm}[htbp]
\caption{FlatQ-Merge Algorithm}
\label{alg:flatq_merge}
\begin{algorithmic}[1]
\REQUIRE Base model $\theta_{\text{base}}$, $K$ target tasks, training datasets $\{D_k\}_{k=1}^K$, SAM perturbation radius $\rho$, quantization bit-width $b$, unlabeled calibration dataset $X = \{x_i\}_{i=1}^N$, test-time learning rate $\eta$, steps $T$.
\ENSURE Optimized adapted merging coefficients $\Lambda^*$, quantized multi-task model $\theta_{\text{quant}}(\Lambda^*, \rho)$.
\STATE \COMMENT{\textbf{Phase 1: Sharpness-Aware Expert Fine-Tuning}}
\FOR{each task $k \in \{1, \dots, K\}$}
    \STATE Fine-tune base model $\theta_{\text{base}}$ on $D_k$ using SAM with radius $\rho$ to obtain expert $\theta_k(\rho)$ (Equation \ref{eq:sam}).
    \STATE Extract layer-wise task vectors: $\tau^l_k(\rho) \leftarrow \theta^l_k(\rho) - \theta^l_{\text{base}}$ for $l \in \{1, \dots, L\}$.
\ENDFOR
\STATE \COMMENT{\textbf{Phase 2: Test-Time Adaptation via Straight-Through PTQ}}
\STATE Initialize merging coefficients $\Lambda^{(0)} \leftarrow \{\lambda^l_k = 0.3\} \in \mathbb{R}^{L \times K}$.
\FOR{step $t = 0, \dots, T-1$}
    \STATE Compute unquantized merged weights for each layer $l \in \{1, \dots, L\}$:
    \STATE $\quad \theta^l_{\text{merged}}\left(\Lambda^{(t)}, \rho\right) \leftarrow \theta^l_{\text{base}} + \sum_{k=1}^K \lambda^l_k \tau^l_k(\rho)$
    \STATE Compute per-channel scale factors $S^l_c\left(\Lambda^{(t)}, \rho\right)$ (Equation \ref{eq:scale}).
    \STATE Quantize and reconstruct weights $\theta^l_{\text{quant}}\left(\Lambda^{(t)}, \rho\right)$ (Equation \ref{eq:quant_round} and \ref{eq:quant}).
    \STATE Pass calibration batch $X$ through the quantized network: $P \leftarrow f\left(X; \theta_{\text{quant}}\left(\Lambda^{(t)}, \rho\right)\right)$.
    \STATE Compute joint prediction entropy: $\mathcal{L}_{\text{entropy}}\left(\Lambda^{(t)}\right) \leftarrow -\frac{1}{N} \sum_{i=1}^N \sum_{j} P_{i,j} \log P_{i,j}$.
    \STATE Compute gradients using Straight-Through Estimator: $\nabla_{\Lambda} \mathcal{L}_{\text{entropy}}\left(\Lambda^{(t)}\right)$.
    \STATE Update coefficients via Adam: $\Lambda^{(t+1)} \leftarrow \text{Adam}\left(\Lambda^{(t)}, \nabla_{\Lambda} \mathcal{L}_{\text{entropy}}\left(\Lambda^{(t)}\right), \eta\right)$.
    \STATE Clip updated coefficients to valid range: $\lambda^l_k \leftarrow \max(0, \min(1, \lambda^l_k))$.
\ENDFOR
\STATE Set $\Lambda^* \leftarrow \Lambda^{(T)}$.
\STATE \textbf{return} $\Lambda^*$, $\theta_{\text{quant}}(\Lambda^*, \rho)$.
\end{algorithmic}
\end{algorithm}
